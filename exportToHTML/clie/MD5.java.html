42       private String digest(String inputStr){
43           byte [] inputBytes=inputStr.getBytes();
44           int byteLen = inputBytes.length; // Длина (байты)
45           int groupCount = 0; // Количество полных групп
46           groupCount = byteLen / 64; // 512 бит на группу (64 байта)
47           long [] groups = null; // каждая группа (64 байта), а затем разделены на 16 групп (4 байта)
48   
49           // Обработка каждой полной группы
50           for(int step=0;step<groupCount;step++){
51               groups=divGroup(inputBytes,step*64);
52               trans (groups); // Группировка обработки, основной алгоритм
53           }
54   
55           // Обрабатываем хвост после полной группировки
56           int rest = byteLen% 64; // остаток после 512-битной группировки
57           byte [] tempBytes=new byte[64];
58           if(rest<=56){
59               for(int i=0;i<rest;i++)
60                   tempBytes[i]=inputBytes[byteLen-rest+i];
61               if(rest<56){
62                   tempBytes[rest]=(byte)(1<<7);
63                   for(int i=1;i<56-rest;i++)
64                       tempBytes[rest+i]=0;
65               }
66               long len=(long)(byteLen<<3);
67               for(int i=0;i<8;i++){
68                   tempBytes[56+i]=(byte)(len&0xFFL);
69                   len=len>>8;
70               }
71               groups=divGroup(tempBytes,0);
72               trans (groups); // группа обработки
73           }else{
74               for(int i=0;i<rest;i++)
75                   tempBytes[i]=inputBytes[byteLen-rest+i];
76               tempBytes[rest]=(byte)(1<<7);
77               for(int i=rest+1;i<64;i++)
78                   tempBytes[i]=0;
79               groups=divGroup(tempBytes,0);
80               trans (groups); // группа обработки
81   
82               for(int i=0;i<56;i++)
83                   tempBytes[i]=0;
84               long len=(long)(byteLen<<3);
85               for(int i=0;i<8;i++){
86                   tempBytes[56+i]=(byte)(len&0xFFL);
87                   len=len>>8;
88               }
89               groups=divGroup(tempBytes,0);
90               trans (groups); // группа обработки
91           }
92   
93           // Преобразуем значение хэша в шестнадцатеричную строку
94           String resStr="";
95           long temp=0;
96           for(int i=0;i<4;i++){
97               for(int j=0;j<4;j++){
98                   temp=result[i]&0x0FL;
99                   String a=hexs[(int)(temp)];
100                  result[i]=result[i]>>4;
101                  temp=result[i]&0x0FL;
102                  resStr+=hexs[(int)(temp)]+a;
103                  result[i]=result[i]>>4;
104              }
105          }
106          return resStr;
107      }

109      /**
110       * Возьмите 512 бит из индекса inputBytes как новый пакет
111       * Разделите каждую 512-битную группу на 16 групп, каждая по 64 бита (8 байтов)
112       * @param inputBytes
113       * @param index
114       * @return
115       */
116      private static long[] divGroup(byte[] inputBytes,int index){
117          long [] temp=new long[16];
118          for(int i=0;i<16;i++){
119              temp[i]=b2iu(inputBytes[4*i+index])|
120                      (b2iu(inputBytes[4*i+1+index]))<<8|
121                      (b2iu(inputBytes[4*i+2+index]))<<16|
122                      (b2iu(inputBytes[4*i+3+index]))<<24;
123          }
124          return temp;
125      }

132      public static long b2iu(byte b){
133          return b < 0 ? b & 0x7F + 128 : b;

241          return y ^ (x | (~z));
